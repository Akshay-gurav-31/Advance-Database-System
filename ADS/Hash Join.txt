-- ORACLE support

-- Create Customers table
CREATE TABLE Customers (
    customer_id INT PRIMARY KEY,
    name VARCHAR(50),
    city VARCHAR(50)
);

-- Create Orders table
CREATE TABLE Orders (
    order_id INT PRIMARY KEY,
    customer_id INT,
    amount INT
);

-- Join Orders with Customers (important join)
SELECT 
    o.order_id,
    o.amount,
    c.name,
    c.city
FROM Orders o
JOIN Customers c
ON o.customer_id = c.customer_id;
